            // Load images only when needed
            class MyImage {
                constructor(src_, backupColor) {
                    this.src = src_;
                    this.img = new Image();

                    this.img.canDraw = false;
                    this.img.drawBackup = false;

                    this.img.onload = ()=>{
                        this.img.canDraw = true;
                    }
                    this.img.onerror = ()=>{
                        this.img.canDraw = false;
                        this.img.drawBackup = true;
                        throw "Unable to load image " + this.src;
                    }
                    this.backupColor = backupColor_;
                    
                }
                draw(x,y,xs,ys) {
                    if(xs == undefined) {
                        xs = this.img.width | 100;
                        ys = this.img.height | 100;
                    }
                    if(this.canDraw) {
                        context.drawImage(this.img, x, y, xs, ys);
                    } else if(this.drawBackup) {
                        context.fillStyle = this.backupColor;
                        context.fillRect(x, y, xs, ys);
                    }
                }
            }
